# Run the Ubuntu 2FA setup script

```bash
# Clone the repository, make the script executable, and run it
git clone https://github.com/LanceAlthouse/everything-scripts.git /tmp/everything-scripts \
  && chmod +x /tmp/everything-scripts/ubuntu/setup_2fa.sh \
  && /tmp/everything-scripts/ubuntu/setup_2fa.sh
```

# Remove script after use
```bash
rm -rf /tmp/everything-scripts
```
